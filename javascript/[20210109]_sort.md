# sort()
- `a > b` return `1`
- `a = b` return `0`
- `a < b` return `-1`

```javascript
log( arr.sort() );
// [1, 100, 20, 34, 9]

/* Ascending order */
let sortArr = arr.sort(function(a, b) { return a - b });
log(sortArr);
// [1, 9, 20, 34, 100]

/* Descending order */
sortArr = arr.sort(function(a, b) { return b - a });
log(sortArr);
// [100, 34, 20, 9, 1]
```
