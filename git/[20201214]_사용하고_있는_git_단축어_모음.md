# 사용하고 있는 git 단축어 모음
```git
alias.co=checkout
alias.cm=commit
alias.cmm=commit -m
alias.cmam=commit --amend
alias.cmamno=commit --amend --no-edit
alias.one=log --pretty=oneline
alias.fuller=log --pretty=fuller
alias.st=status
alias.pop=stash pop
alias.ba=branch -a
alias.p=pull origin
alias.po=push origin
alias.cp=cherry-pick
alias.rb=rebase
alias.rbc=rebase --continue
```
